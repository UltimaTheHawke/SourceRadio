CREATE TABLE tftunes.SONG_REQUEST (
ID INT NOT NULL AUTO_INCREMENT ,
Timestamp TIMESTAMP NOT NULL ,
Query VARCHAR(125) NOT NULL ,
SongID VARCHAR(11) NULL ,
Username VARCHAR(125) NOT NULL ,
WasAdmin BOOLEAN NOT NULL ,
WasOwner BOOLEAN NOT NULL ,
Extended BOOLEAN NOT NULL ,
Skipped BOOLEAN NOT NULL ,
UsedCachedQuery BOOLEAN NOT NULL,
UsedCachedAudio BOOLEAN NOT NULL,
CONSTRAINT REQUEST_PK PRIMARY KEY(ID),
CONSTRAINT REQUEST_SONG_FK FOREIGN KEY(SongID)
	REFERENCES SONG(YoutubeID)
     		ON UPDATE CASCADE
		ON DELETE RESTRICT,
UNIQUE (ID)) ENGINE = InnoDB;
